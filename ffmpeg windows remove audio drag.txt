@echo off
REM Drag and drop video file(s) onto this script to remove audio

if "%~1"=="" (
    echo Drag and drop video file(s) onto this script.
    pause
    exit /b
)

:loop
set "input=%~1"
set "name=%~n1"
set "ext=%~x1"
set "output=%name%_noaudio%ext%"

echo Processing "%input%"...
ffmpeg -i "%input%" -c copy -an "%output%"

shift
if not "%~1"=="" goto loop

echo.
echo Done! Created file(s) with "_noaudio" suffix.
pause
